19. find duplicate values in array:

def remove_duplicate(list):
    lis = []
    for i in list:
        if i not in lis:
            lis.append(i)
    return lis

list = [2,4,3,5,3,2,8,8,4,9,3,4]
print(remove_duplicate(list))

20. print duplicate values in array:

def remove_duplicate(list):
    lis = []
    for i in list:
        if list.count(i) > 1 and i not in lis:
            lis.append(i)
    return lis

list = [2,4,3,5,3,2,8,8,4,9,3,4]
print(remove_duplicate(list))
